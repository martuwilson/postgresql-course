SELECT
	name,
	SUBSTRING(name, 0, 5),
	position(' ' IN name),
	SUBSTRING(name, 0, position(' ' IN name)) as first_name,
	TRIM(SUBSTRING(name, position(' ' IN name))) as last_name
from users;
